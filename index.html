                        private:
                            int ** matrix;
                            int ver;
                        public:
                            Graph(int ver) {
                                this->ver = ver;
                                matrix = new int* [ver];
                                for (int i = 0; i < ver; i++) {
                                    matrix[i] = new int [ver];
                                    for (int j = 0; j < ver; j++) matrix[i][j] = 0;
                                }
                            }
                      }
